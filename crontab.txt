SHELL=/bin/bash
PATH=/usr/local/bin:/usr/bin:/bin
# All times UTC. CST = UTC+8.
# Every row goes through scripts/cron_wrap.sh which provides:
#   - per-slot dated logs (logs/pepperbot-<slot>-YYYY-MM-DD.log)
#   - flock contention is logged + webhook-alerted (NOT silent)
#   - non-zero exit fires webhook alert with last 10 log lines
#   - 14-day log retention

# 左手 7x24 学习管道：
#   observe = 拉 KOL 推 + guba 文章 (raw_url 去重，guba 多拉零成本)
#   mine    = 把 tier>=1 的 KOL 观察蒸馏成 technique_entries
# 30 分钟一次 observe = 48 次/天，x_list_finance 12/h 配额绰绰有余
# 2 小时一次 mine = 12 次/天，让冷启动 50 阈值在 2-3 天内跨过
*/30 * * * * root /app/scripts/cron_wrap.sh observe    /app/scripts/run.sh observe
0 */2 * * *  root /app/scripts/cron_wrap.sh mine       /app/scripts/run.sh mine

# 右手生产：morning batch 23:00 UTC = 07:00 CST，生成 20 条草稿推 Discord。
0 23 * * *   root /app/scripts/cron_wrap.sh morning_batch /app/scripts/run.sh batch_post

# Discord 审批 gate 每 5 分钟轮询一次。
*/5 * * * *  root /app/scripts/cron_wrap.sh discord_poll  /app/scripts/run.sh discord_poll

# 日终 review (16:00 UTC = 00:00 CST)：评分昨日推文 + EMA 更新技巧成功分。
# mine 已经独立跑了，这里 review 单飞。
0 16 * * *   root /app/scripts/cron_wrap.sh nightly       /app/scripts/run.sh review

# 周日 04:00 UTC = 12:00 CST 全量 remine：重算所有边 + 衰减 + 剪枝。
0 4 * * 0    root /app/scripts/cron_wrap.sh remine        /app/scripts/run.sh remine
