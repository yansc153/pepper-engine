SHELL=/bin/bash
PATH=/usr/local/bin:/usr/bin:/bin
# All times UTC. CST = UTC+8.
# Every row goes through scripts/cron_wrap.sh which provides:
#   - per-slot dated logs (logs/pepperbot-<slot>-YYYY-MM-DD.log)
#   - flock contention is logged + webhook-alerted (NOT silent)
#   - non-zero exit fires webhook alert with last 10 log lines
#   - 14-day log retention

# Morning batch: 23:00 UTC = 07:00 Beijing CST. Observe 20 articles then
# generate up to 20 drafts, all pushed to Discord for user review. User
# wakes up to 20 drafts ready to triage.
30 22 * * * root /app/scripts/cron_wrap.sh morning_observe /app/scripts/run.sh observe
0 23 * * *  root /app/scripts/cron_wrap.sh morning_batch   /app/scripts/run.sh batch_post

# Discord approval gate poller every 5 min (so user reactions are processed
# throughout the day).
*/5 * * * *          root /app/scripts/cron_wrap.sh discord_poll  /app/scripts/run.sh discord_poll

# Self-monitor (cross-reference wild posts) every 6h
0 */6 * * *          root /app/scripts/cron_wrap.sh self_monitor  /app/scripts/run.sh self_monitor

# Daily mine -> review at 16:00 UTC = 00:00 CST (chained under one lock)
0 16 * * *           root /app/scripts/cron_wrap.sh nightly       bash -c '/app/scripts/run.sh mine && /app/scripts/run.sh review'

# Weekly full remine: Sunday 04:00 UTC = 12:00 CST Sun
0 4 * * 0            root /app/scripts/cron_wrap.sh remine        /app/scripts/run.sh remine
